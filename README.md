# Clone the project
git clone https://github.com/AlexTMIT/pixelator-studio.git

# Go into the folder
cd pixelator-studio

# Create virtual environment
python3 -m venv venv

# Activate it
source venv/bin/activate

# Install dependencies
pip install -r requirements.txt

# Run the app
python main.py
